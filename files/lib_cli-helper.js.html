  /**
   * @private
   * @method parseDate
   * @static
   * @param date *
   * @returns {Date}
   */
  parseDate      : function (date) {
    var split = date.split(":");
    if (split.length === 3) {
      var InputDate = new Date(split[2], split[1] - 1, split[0]);
      if (!isNaN(InputDate)) {
        //console.log("Input Date is " + InputDate.getDate() + " "  + InputDate.getMonth() + " " + InputDate.getYear());
        console.log("Date is " + InputDate);
        return InputDate;
      }
    }
    console.log("Invalid Date Format. Please enter Date in the format DD:MM:YYYY");
  }
